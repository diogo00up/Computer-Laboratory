326void (game_mouse_click)(uint16_t x, uint16_t y) {
327 if(game_state == SCORES) {
328 if(x > (hres/2)-350 && x < (hres/2)-350 + 200 && y < vres/2 + 125 + 90 && y > vres/2 + 125) {
329 game_state = MENU;
330 }
331 return;
332 }
333
334 if(x > (hres/2)-350 && x < (hres/2)-350 + 200 && y < vres/2 + 100 + 90 && y > vres/2 + 100) {
335 game_state = IN_GAME;
336 counter = 0;
337 score = 0;
338 lives_pacman = 3;
339 reset_board();
340 draw_background();
341 draw_coin();
342 draw_pacman();
343 draw_score();
344 draw_lives();
345 move_buffer();
346 return;
347 }
348 if(x > (hres/2)-100 && x < (hres/2)-100 + 200 && y < vres/2 + 100 + 90 && y > vres/2 + 100) {
349 game_state = SCORES;
350 for (int i = 0; i < 6; i++) {
351 for (int j = i + 1; j < 6; j++) {
352 if (high_scores[i] < high_scores[j]) {
353 int aux = high_scores[i];
354 high_scores[i] = high_scores[j];
355 high_scores[j] = aux;
356 }
357 }
358 }
359 return;
360 }
361 if(x > (hres/2)+150 && x < (hres/2)+150 + 200 && y < vres/2 + 100 + 90 && y > vres/2 + 100) {
362 scancode=KEY_ESC;
363 return;
364 }

634int (game_loop)() {
635 draw_menu(selected);
636 draw_mouse();
637 draw_score();
638 move_buffer();
639
640
641 int ipc_status;
642 message msg;
643 int r;
644
645 while (scancode != KEY_ESC) {
646 if ((r = driver_receive(ANY, &msg, &ipc_status)) != 0) {
647 printf("driver_receive failed with: %d", r);
648 continue;
649 }
650 if (is_ipc_notify(ipc_status)) {
651 switch (_ENDPOINT_P(msg.m_source)) {
652 case HARDWARE:
653 if (msg.m_notify.interrupts & bit_no_keyboard) {
654 kbc_ih();
655 if(game_state == MENU){
656 if (scancode==DOWN || scancode==LEFT || scancode==A) {
657 selected--;
658 }
659 if (scancode==UP || scancode== RIGHT || scancode==D) {
660 selected++;
661 }
662 if (selected==3) {
663 selected = 0;
664 }
665 if (selected==-1) {
666 selected = 2;
667 }
668 if (scancode==ENTER && selected==0) {
669 game_state = IN_GAME;
670 counter = 0;
671 score = 0;
672 lives_pacman = 3;
673 reset_board();
674 draw_background();
675 draw_coin();
676 draw_pacman();
677 draw_score();
678 draw_lives();
679 move_buffer();
680 }
681 if (scancode==ENTER && selected==1) {
682 for (int i = 0; i < 6; i++) {
683 for (int j = i + 1; j < 6; j++) {
684 if (high_scores[i] < high_scores[j]) {
685 int aux = high_scores[i];
686 high_scores[i] = high_scores[j];
687 high_scores[j] = aux;
688 }
689 }
690 }
691 game_state = SCORES;
692 }
693 if (scancode==ENTER && selected==2) {
694 scancode=KEY_ESC;
695 }
696
697 } else if(game_state == IN_GAME) {
698 switch (scancode){
699 case D:
700 pacman.moving = true;
701 if (pacman.dir != right) {
702 dir_buffer = right;
703 }
704 break;
705 case A:
706 pacman.moving = true;
707 if (pacman.dir != left) {
708 dir_buffer = left;
709 }
710 break;
711 case W:
712 pacman.moving = true;
713 if (pacman.dir != up) {
714 dir_buffer = up;
715 }
716 break;
717 case S:
718 pacman.moving = true;
719 if (pacman.dir != down) {
720 dir_buffer = down;
721 }
722 break;
723 case Q:
724 reset_pacman_ghost_values();
725 for (int i = 0; i < 5; i++) {
726 for (int j = i + 1; j < 5; j++) {
727 if (high_scores[i] < high_scores[j]) {
728 int aux = high_scores[i];
729 high_scores[i] = high_scores[j];
730 high_scores[j] = aux;
731 }
732 }
733 }
734 high_scores[5] = score;
735 game_state = MENU;
736 break;
737
738 default:
739 break;
740 }
741 } else if(game_state == SCORES) {
742 if (scancode==ENTER) {
743 game_state = MENU;
744 }
745 }
746 }
747
748 if (msg.m_notify.interrupts & bit_no_timer) {
749 timer_int_handler();
750 if(game_state == MENU){
751 draw_menu(selected);
752 draw_mouse();
753 move_buffer();
754 } else if(game_state == IN_GAME) {
755 pickup_coins();
756 if (score == 244) game_state = MENU;
757 if (GP_collisions(&pink_ghost)){
758 lives_pacman--;
759 reset_pacman_ghost_values();
760 }
761 if (GP_collisions(&red_ghost)) {
762 lives_pacman--;
763 reset_pacman_ghost_values();
764 }
765 if (GP_collisions(&orange_ghost)) {
766 lives_pacman--;
767 reset_pacman_ghost_values();
768 }
769 if (GP_collisions(&blue_ghost)) {
770 lives_pacman--;
771 reset_pacman_ghost_values();
772 }
773 if(lives_pacman == 0) {
774 reset_pacman_ghost_values();
775 for (int i = 0; i < 5; i++) {
776 for (int j = i + 1; j < 5; j++) {
777 if (high_scores[i] < high_scores[j]) {
778 int aux = high_scores[i];
779 high_scores[i] = high_scores[j];
780 high_scores[j] = aux;
781 }
782 }
783 }
784 high_scores[5] = score;
785 game_state = MENU;
786 break;
787 }
788 if(pacman.pos_counter == 0 && pacman.yBPos == 13){
789 if (pacman.xBPos == 0){
790 pacman.xBPos = 25;
791 pacman.xPos = 102 + 575;
792 } else if (pacman.xBPos == 25){
793 pacman.xBPos = 0;
794 pacman.xPos = 102;
795 }
796 }
797 if(pacman.pos_counter == 0 && !collision(dir_buffer)) pacman.dir = dir_buffer;
798
799 if(pacman.moving) {
800 if(red_ghost.jail_timer == 0) ghost_movement(&red_ghost);
801 if (counter/60.0 >= 10){
802 if (!pink_ghost.free && pink_ghost.jail_timer == 0){
803 pink_ghost.xPos = 102 + 299;
804 pink_ghost.yPos = 25 + 190;
805 pink_ghost.free = true;
806 }
807 ghost_movement(&pink_ghost);
808 }
809 if (counter/60.0 >= 20){
810 if (!blue_ghost.free && blue_ghost.jail_timer == 0){
811 blue_ghost.xPos = 102 + 299;
812 blue_ghost.yPos = 25 + 190;
813 blue_ghost.free = true;
814 }
815 ghost_movement(&blue_ghost);
816 }
817 if (counter/60.0 >= 30){
818 if (!orange_ghost.free && orange_ghost.jail_timer == 0){
819 orange_ghost.xPos = 102 + 299;
820 orange_ghost.yPos = 25 + 190;
821 orange_ghost.free = true;
822 }
823 ghost_movement(&orange_ghost);
824 }
825 switch (pacman.dir) {
826 case down:
827 if (!collision(pacman.dir)){
828 pacman.yPos += pacman.speed;
829 pacman.pos_counter += pacman.speed;
830 if (pacman.pos_counter % 19 == 0){
831 pacman.pos_counter = 0;
832 pacman.yBPos++;
833 }
834 }
835 break;
836 case up:
837 if (!collision(pacman.dir)){
838 pacman.yPos -= pacman.speed;
839 pacman.pos_counter += pacman.speed;
840 if (pacman.pos_counter % 19 == 0){
841 pacman.pos_counter = 0;
842 pacman.yBPos--;
843 }
844 }
845 break;
846 case left:
847 if (!collision(pacman.dir)){
848 pacman.xPos -= pacman.speed;
849 pacman.pos_counter += pacman.speed;
850 if (pacman.pos_counter % 23 == 0){
851 pacman.pos_counter = 0;
852 pacman.xBPos--;
853 }
854 }
855 break;
856 case right:
857 if (!collision(pacman.dir)){
858 pacman.xPos += pacman.speed;
859 pacman.pos_counter += pacman.speed;
860 if (pacman.pos_counter % 23 == 0){
861 pacman.pos_counter = 0;
862 pacman.xBPos++;
863 }

865 break;
866 default:
867 break;
868 }
869 }

871 draw_background();
872 draw_coin();
873 if (red_ghost.jail_timer == 0) draw_ghost(&red_ghost);
874 if (pink_ghost.jail_timer == 0) draw_ghost(&pink_ghost);
875 if (blue_ghost.jail_timer == 0) draw_ghost(&blue_ghost);
876 if (orange_ghost.jail_timer == 0) draw_ghost(&orange_ghost);
877 draw_pacman();
878 draw_score();
879 draw_lives();
880 draw_mouse();
881 move_buffer();
882 } else if(game_state == SCORES) {
883 draw_scores_menu();
884 draw_mouse();
885 move_buffer();
886 }
887
888 }
889
890 if (msg.m_notify.interrupts & bit_no_mouse) {
891 mouse_ih();
892 assemble_mouse_packet();
893 if(mouse_counter == 3) {
894 parse_mouse_packet();
895 mouse.xPos += mouse_packet.delta_x;
896 mouse.yPos -= mouse_packet.delta_y;
897 if(mouse.xPos > hres - mouse.img.width) {
898 mouse.xPos = hres - mouse.img.width;
899 } else if(mouse.xPos < 0) {
900 mouse.xPos = 0;
901 }
902
903 if(mouse.yPos > vres - mouse.img.height) {
904 mouse.yPos = vres - mouse.img.height;
905 } else if(mouse.yPos < 0) {
906 mouse.yPos = 0;
907 }
908 if(game_state == MENU || game_state == SCORES) {
909 game_mouse_hover(mouse.xPos, mouse.yPos);
910 if(mouse_packet.lb) {
911 game_mouse_click(mouse.xPos, mouse.yPos);
912 }
913 }
914 }
915 }
916
917 break;
918 default:
919 break;
920 }
921 }
922 }
923 return 0;
924}
